        //-----------------------------------
        //  jQuery & jQueryRotate functions 
        //----------------------------------

        $(document).ready(function () {

            $('#welcome').on('click', function () {

                $('#wlogo').hide(function () {

                    window.location.href = 'sts.php';

                });

            });


            $('#welcome').on('mouseover', function () {

                var _currentAngle;

                _currentAngle = $('#wlogo').getRotateAngle();

                if ((_currentAngle == '0') || (_currentAngle == '')) {

                        $('#wlogo').rotate({

                            animateTo: 360,

                            callback: function () {

                                $('#wlogo').rotate({
                                    angle: 0
                                });

                            }
                        });

                };

            });


        });
    </script>
